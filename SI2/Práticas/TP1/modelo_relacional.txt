Modelo Relacional:

Cliente(IDCliente[AK], Contacto, Nome)
IDCliente - int @@IDENTITY
Contacto - short
Nome - varchar(128)

Comentario(IDComentario[AK], IDCliente[FK], Nota, Observacoes)
IDComentario - int @@IDENTITY
Nota - int de 0 a 5
Observacoes - varchar(256)

Reuniao(Data[AK], Conclusao, IDComentario[FK])
Restricoes:
Data - datetime > actual
Conclusao - varchar(256)
IDComentario - int

CLienteRegistado(IDCliente[AK][FK], Endereco_Postal, Nacionalidade)
IDCliente - int
Endereco_Postal - varchar(56)
Nacionalidade - varchar(56)

Amigos(IDCliente[AK][FK], IDAmigo[AK][FK])
IDCliente - int
IDAmigo - int

Cardapio(IDCardapio[AK], Descricao, Categoria)
IDCardapio - int @@IDENTITY
Descricao - varchar(256)
Categoria - varchar(256)

ReservaNormal(IDReserva[AK], IDCardapio[FK], Data, QtdPessoas)
IDReserva - int @@IDENTITY
Data - datetime
QtdPessoas - int

ReservaEvento(IDReserva[AK][FK], Organizador[FK], Designacao, Descricao)
IDReserva - int
Organizador - int //Organizador - ID do Cliente
Designacao - varchar(128)
Descricao - varchar(256)

Cli_R(IDCliente[AK][FK],IDReserva[AK][FK])
IDCliente - int
IDReserva - int

Pratos(IDPrato[AK], Preco, Qtd, Activo)
IDPrato- int @@IDENTITY
Preco - decimal(12,5)
Qtd - smallint
activo - bit

Car_P(IDCardapio[AK][FK], IDPrato[AK][FK])
IDCardapio - int
IDPrato - int

Ingredientes(IDIngrediente[AK], Nome, Descricao, Qtd_Reserva, Qtd_Actual)
IDIngrediente - int @@IDENTITY
Nome - varchar(128)
Descricao - varchar(256)
Qtd_Reserva - int
Qtd_Actual - int

P_I(IDPrato[AK][FK], IDIngrediente[AK][FK])
IDPrato - int
IDIngrediente - int

Fornecedores(IDFornecedor[AK], Nome, Morada, Codigo_Postal, Pais)
IDFornecedor - int @@IDENTITY
Nome - varchar(128)
Morada - varchar(256)
Codigo_Postal - varchar(56)
Pais - varchar(56)
 
Enc(IDEncomendas[AK][FK], IDFornecedor[AK][FK], IDIngrediente[AK][FK], Qtd, Data, Data_Entrega)
IDEncomendas - int @@IDENTITY

IDIngrediente - int
